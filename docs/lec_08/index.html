Семантика перемещения

Проблема

String GetData(tcp::socket socket);
void Modify(String& str);
std::vector<String> storage;

for (size_t i = 0; i < N; ++i) {
  auto data = GetData(socket);
  Modify(data);
  storage[i] = data; // Копирование строки.
} // Вызывается деструктор |data|.

We need to go deeper

Копирование строк

// string.h
class String {
  char* data_;
 public:
  String(const String& oth);
  String& operator=(const String& oth);
};

// string.cpp
String::String(const String& oth) {
  data_ = new char[oth.size() + 1];
  memcpy(data_, oth.data_, oth.size() + 1);
}

String& String::operator=(const String& oth) {
  if (this != &oth) {
    delete[] data_;
    data_ = new char[oth.size() + 1];
    memcpy(data_, oth.data_, oth.size() + 1);
  }
  return *this;
}

Идея

String::move_to(String& b) {
  char* tmp = this->data_;
  this->data_ = b.data_;
  b.data_ = tmp;
}

for (size_t i = 0; i < N; ++i) {
  auto data = GetData(socket);
  Modify(data);
  data.move_to(storage[i]); // Избавляемся от копирования.
} // Вызывается деструктор пустой |data|.

Move-семантина С++03

for (size_t i = 0; i < N; ++i) {
  auto data = GetData(socket);
  Modify(data);
  data.swap(storage[i]); // Избавляемся от копирования.
} // Вызывается деструктор пустой |data|.

swap как move-семантика

Использование swap помогает повысить производительность приложения.

Однако, иногда использовать swap для перемещения данных неудобно. Во-вторых, это усложняет и запутывает код.

Move-семантина С++11

for (size_t i = 0; i < N; ++i) {
  auto data = GetData(socket);
  Modify(data);
  storage[i] = std::move(data); // Избавляемся от копирования.
} // Вызывается деструктор пустой |data|.

Перемещение строк

// string.h
class String {
  char* data_;
 public:
  String(String&& oth);
  String& operator=(String&& oth);
};

// string.cpp
String::String(String&& oth) {
  data_ = oth.data_;
  oth.data_ = nullptr;
}

String& String::operator=(String&& oth) {
  if (this != &oth) {
    delete[] data_;
    data_ = oth.data_;
    oth.data_ = nullptr;
  }
  return *this;
}

rvalue

Rvalue ссылки – техническое расширение языка C++. Они позволяют компилятору определить, когда необходимо использовать перемещение, вместо копирования.

Перегрузка функций

// Копирование.
String(const String& oth);
String& operator=(const String& oth);

// Перемещение.
String(String&& oth);
String& operator=(String&& oth);

Благодаря rvalue, компиятор определяет какую из перегрузок функций необходимо выбрать.

std::move()

std::move безусловно приводит аргумент к rvalue ссылке.

template<typename T>
decltype(auto) move(T&& arg) {
  using ReturnType = remove_reference_t<T>&&;
  return static_cast<ReturnType>(arg);
}

С помощью функции std::move разработчик “подсказавает” компилятору, что необходимо использовать перемещение объекта.

Рекомендации

  • Не используйте объект после того, как его переместили в другой!
  • Не объявляйте объекты константными, если хотите иметь возможность их перемещать
  • std::move не гарантирует, что приведенный объект будет иметь право быть перемещенным

Резюме

  • Семантика перемещения позволяет компиляторам заменять дорогостоящие операции копирования “дешевыми” перемещениями.
  • Конструктор перемещения и оператор перемещения предоставляют контроль над семантикой перемещения.

Резюме

  • Семантика перемещения позволяет также создавать типы, которые концептуально поддерживают только операции перемещения. К таким типам относятся как std::unique_ptr, std::future, std::thread и др.

Шаблоны проектирования

Что это

Шаблон проектирования – повторяемая архитектурная конструкция, представляющая собой решение проблемы проектирования в рамках некоторого часто возникающего контекста.

Зачем нужны шаблоны

  • предоставляют готовое решение некоторой известной проблемы
  • позволяют поддерживать архитектуру приложения в лучшем состоянии, нежели без использования шаблонов
  • помогают разговаривать разраотчикам на одном языке
  • полезны при рефакторинге кода

Неоправданное применение

Самая большая ошибка при работе с шаблонами проектирования – это неоправданное их использование.

Если у тебя в руках молоток, то все предметы вокруг начинают напоминать гвозди.

Простая фабрика

В С++ существует механизм полиморфизма на базе виртуальных функций.

Полиморфизм позволяет вызывать функции производным классов, ничего не зная о них, а только оперировать базовым классом.

Полиморфизм

struct Unit {
  void Attack(Unit& target);
}

struct Knight : public Unit {
  // ...
};

struct Archer : public Unit {
  // ...
};

Unit* unit = new Knight();
Unit* target = new Archer();
unit->Attack(*target);

Виртуальный конструктор

Однако, чтобы создать объект дочернего класса необходимо явно указать в коде, объект какого класса создается.

Решить данную проблему призван шаблон проектирования “простая фабрика”

Простая фабрика

Паттерн “простая фабрика” – это класс, в котором есть один метод с большим условным оператором, выбирающим создаваемый продукт.

Этот метод вызывают с неким параметром, по которому определяется какой из продуктов нужно создать.

Реализация

struct Barrack {
  Unit* CreateUnit(UnitID unit_id) {

    switch (unit_id) {
      case UnitID::Knight:
        return new Knight();
      case UnitID::Archer:
        return new Archer();
      default:
        return nullptr;
    }

  }
};

Фабричный метод

Паттерн “фабричный метод” – это устройство классов, при котором подклассы могут переопределять тип создаваемого в суперклассе продукта.

Пример

struct Barrack {
  virtual Unit* CreateUnit() = 0;

  void Hire(Player& player) {
    auto* unit = CreateUnit();
    player.AddToArmy(unit);
  }
};

struct KnightBarrack : public Barrack {
  Unit* CreateUnit() override {
    return new Knight();
  }
};

Пример (продолжение)

struct ArcherBarrack : public Barrack {
  Unit* CreateUnit() override {
    return new Archer();
  }
};

ArcherBarrack archer_barrack;
archer_barrack.Hire(player);

Singleton

Singleton – порождающий шаблон проектирования, гарантирующий, что в приложении будет единственный экземпляр некоторого класса, и предоставляющий глобальную точку доступа к этому экземпляру.

Назначение паттерна

Некоторые подсистемы в приложении должны существовать только в единственном экземпляре.

Например, система логгирования или драйвер дисплея.

Цели паттерна

  1. Гарантирует наличие единственного экземпляра класса

  2. Предоставляет глобальную точку доступа

Проблемы паттерна

С данным паттером традиционно связаны несколько проблем:

  • доступ и создание объекта в многопоточной среде
  • время создания и разрушения объекта singleton’a

Классическая реализация Singleton

// Singleton.h
class Singleton {
  static Singleton* instance;
  Singleton() {}
  Singleton(const Singleton&) = delete;
  Singleton& operator=(Singleton&) = delete;
public:

  static Singleton * getInstance() {
    if(!instance)
      instance = new Singleton();
    return instance;
  }
};

// Singleton.cpp
#include "Singleton.h"

Singleton* Singleton::instance = nullptr;

Реализация. Синглтон Мэйерса

// Singleton.h
class Singleton {
private:
  Singleton() {}
  Singleton(const Singleton&) = delete;
  Singleton& operator=(Singleton&) = delete;
public:
  static Singleton& Instance() {
    static Singleton instance;
    return instance;
  }
};

Observer

Наблюдатель – это поведенческий паттерн проектирования, который создаёт механизм подписки, позволяющий одним объектам следить и реагировать на события, происходящие в других объектах.

Интерфейс Publisher

struct Publisher {
  struct Observer {
    virtual void OnUpdate() = 0;
  };
  void AddObserver(const Observer& observer);
  void RemoveObserver(const Observer& observer);
  void DoSomething();
 private:
  void Notify();
  std::vector<Observer*> observers_;
};

Реализация Publisher

void Publisher::AddObserver(const Observer& observer) {
  observers_.push_back(&observer);
}

void Publisher::RemoveObserver(const Observer& observer) {
  auto it = std::remove(
      observers_.begin(), observers_.end(), &observer);
  observers_.erase(it, observers_.end());
}

Реализация Publisher

void Publisher::Notify() {
  for (auto* observer : observers_) {
    observer->OnUpdate();
  }
}

void Publisher::DoSomething() {
  // Выполнение некоторых действий.
  // Решаем, что необходимо умедомить наблюдателей о событии.
  Notify();
  // Продолжчаем выполнение функции.
}

Использование Observer

struct Subscriber : public Publisher::Observer {
  void OnUpdate() {
    // ...
  }
};

Publisher publisher;
Subscriber subscriber;
publisher.AddObserver(subscriber);
publisher.DoSomething();

STL

STL – набор согласованных обобщённых алгоритмов, контейнеров, средств доступа к их содержимому и различных вспомогательных функций в C++

Алгоритмы STL

Алгоритмы STL реализованы в виде глобальных функций, которые работают с использованием итераторов.

Это означает, что реализованный алгоритм будет работать со всеми контейнерами, которые предоставляют набор итераторов.

Некоторые из алгоритмов

Алгоритм Описание
count_if подсчет элементов
find_if нахождение элемента
transform преобразование диапазона
all_of все элементы удовлетворяют условию
remove_if удаление из диапазона
replace_if замена элементов
merge слияние сортированных диапазонов
is_sorted сортирован ли диапазон
lower_bound возвращает позицию на элемент, который не меньше чем аргумент

Разбор аргументов

auto it = std::find_if(
  v.begin(), // начало диапозона.
  v.end(), // конец диапозона.
  [](const auto& el) { // предикат.
    return el.size() > 5;
  }
)

Сначала задается диапазон [first, last), элемент last не включается в диапазон. Далее идет предикат применяющийся к диапазону.

Современный С++

std::optional

Класс std::optional управляет опциональным значением, т. е. значением, которое может присутствовать или отсутствовать.

Пример

std::optional<int> TryParseNumber(const std::string& str);

auto num = TryParseNumber("123");
if (num) {
  std::cout << num.value(); // или *num;
}
std::cout << num.value_or(0);

auto num = TryParseNumber("Not number");
if (num) {
  std::cout << num.value(); // или *num;
}
std::cout << num.value_or(0);

std::any

Класс std::any – типобезопасносный контейнер для одиночного значения любого типа.

Пример

std::any value;

value = 1;
value = std::string("Something");
value = Student{"Ivanov", "BMSTU", 2019};
std::string name = std::any_cast<Student>(value).Name;

std::tuple

Класс std::tuple представляет собой набор фиксированного размера различный типов.

std::tuple<std::string, std::string, int, float> value;
value = std::make_tuple("ABC", "qwerty", 1, 4.3);

std::string first;
std::string second;
int third = 0;
float fourth = 0;
std::tie(first, second, third, fourth) = value;

std::cout << first << second << third << fourth;

std::variant

Класс std::variant представляет собой типобезопасное объединение (union).

std::variant<int, float> v;
v = 12
int a = std::get<int>(v);
int b = std::get<0>(v);

try {
  std::get<float>(v);
}
catch (const std::bad_variant_access&) {}

Структурное связывание

До
std::map<std::string, int> students = GetGtoupById(31);

for (const auto& e : students) {
  std::cout << e.first << " has mark = " << e.second;
}
После
std::map<std::string, int> students = GetGtoupById(31);

for (const auto& [name, mark] : students) {
  std::cout << name << " has mark = " << mark;
}
